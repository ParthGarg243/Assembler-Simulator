mov R1 $3
mov R2 $2
mov R3 $1
mov R0 $0

outer_loop_start:
    cmp R1 $0
    je outer_loop_end

    middle_loop_start:
        cmp R2 $0
        je middle_loop_end

        inner_loop_start:
            cmp R3 $0
            je inner_loop_end

            swapf R0, R1
            bsf R1 $3
            bcf R2 $2
            rrf R0 $1
            rlf R3 $1

            sub R3 $1
            jmp inner_loop_start

        inner_loop_end:
        mov R3 $1

        sub R2 $1
        jmp middle_loop_start

    middle_loop_end:
    mov R2 $2

    sub R1 $1
    jmp outer_loop_start

outer_loop_end:
hlt
